    def get_path(self, account_id: int) -> str:
        """Get account path with caching."""
        if account_id in self._path_cache:
            return self._path_cache[account_id]

        # –ö—ç—à–∏—Ä—É–µ—Ç –ø–æ—Å—Ç—Ä–æ–µ–Ω–∏–µ –ø—É—Ç–∏
